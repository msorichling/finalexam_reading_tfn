    <script>
        // #######################################################
        // 1. ÜBUNGSDATEN (Die Daten bleiben unverändert)
        // #######################################################
        
        const tfData = [
            { id: 1, statement: "The Post interviewed Wintersoul and Dellis about the documentary presenting them.", answer: "F" },
            { id: 2, statement: "Brain techniques helped Wintersoul succeed at university.", answer: "T" },
            { id: 3, statement: "Information gets processed from one type of memory to another.", answer: "T" },
            { id: 4, statement: "Our working memory stores information for less than one minute.", answer: "N" },
            { id: 5, statement: "According to Cowan, the human brain’s storage capacity is limited.", answer: "F" },
            { id: 6, statement: "Teen and adult brains rely on different systems to store information.", answer: "N" },
            { id: 7, statement: "When in a traffic jam, Dellis practises remembering people’s names.", answer: "F" },
            { id: 8, statement: "When Wintersoul was booked for a commercial, she increased the challenge.", answer: "T" },
            { id: 9, statement: "Positive associations helped Wintersoul remember information.", answer: "T" },
            { id: 10, statement: "The processes of learning and of recalling work best in similar situations.", answer: "T" }
        ];

        // #######################################################
        // 2. RENDERING & LOGIK
        // #######################################################
        
        function renderTFGrid() {
            const grid = document.getElementById('tfGrid');
            const itemsHtml = tfData.map(item => `
                <div class="tf-item" id="item-${item.id}">
                    <span class="tf-number">${item.id}.</span>
                    <span class="tf-statement">${item.statement}</span>
                    <span class="tf-options">
                        <input type="radio" id="q${item.id}-t" name="q${item.id}" value="T">
                        <input type="radio" id="q${item.id}-f" name="q${item.id}" value="F">
                        <input type="radio" id="q${item.id}-n" name="q${item.id}" value="N">
                    </span>
                    <span id="tf-solution-${item.id}" class="correct-answer">(Korrekt: ${item.answer})</span>
                </div>
            `).join('');

            // Append items below the header
            grid.insertAdjacentHTML('beforeend', itemsHtml);
        }

        function checkAllTasks() {
            let correctCount = 0;
            let totalCount = tfData.length;

            tfData.forEach(item => {
                const radios = document.getElementsByName(`q${item.id}`);
                let userAnswer = '';
                
                // Find user selection
                for (const radio of radios) {
                    // Reset styling on all options for this question
                    radio.style.boxShadow = 'none';

                    if (radio.checked) {
                        userAnswer = radio.value;
                    }
                }
                
                // Check answer and apply coloring
                if (userAnswer === item.answer) {
                    correctCount++;
                    // Highlight the correct radio button in green
                    document.getElementById(`q${item.id}-${userAnswer.toLowerCase()}`).style.boxShadow = `0 0 0 3px var(--color-success)`;
                } else if (userAnswer !== '') {
                    // If wrong answer was selected, highlight incorrect selection in red
                    document.getElementById(`q${item.id}-${userAnswer.toLowerCase()}`).style.boxShadow = `0 0 0 3px var(--color-error)`;
                    // Optionally highlight the correct answer in green
                    document.getElementById(`q${item.id}-${item.answer.toLowerCase()}`).style.boxShadow = `0 0 0 3px var(--color-success)`;
                }
            });
            
            const scorePercentage = (correctCount / totalCount) * 100;
            const resultMessage = document.getElementById('resultMessage');
            resultMessage.style.display = 'block';

            let feedback;
            if (scorePercentage === 100) {
                feedback = `**Perfekt!** Alle ${totalCount} Aussagen sind korrekt.`;
            } else if (scorePercentage >= 80) {
                feedback = `**Sehr gut!** ${correctCount} von ${totalCount} korrekt. Du hast die meisten Details erfasst.`;
            } else if (scorePercentage >= 60) {
                feedback = `**Gut.** ${correctCount} von ${totalCount} korrekt. Bitte vergleiche deine Antworten mit der Lösungsansicht.`;
            } else {
                feedback = `**Weiter üben.** ${correctCount} von ${totalCount} korrekt. Nutze die Lösungsansicht, um die Punkte im Text zu finden.`;
            }

            resultMessage.innerHTML = feedback.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>'); 

            if (correctCount === totalCount) {
                resultMessage.className = 'result-message success';
            } else {
                resultMessage.className = 'result-message failure';
            }
        }
        
        function toggleSolutions() {
            const solutionSpans = document.querySelectorAll('.tf-item .correct-answer');
            const button = document.getElementById('solutionButton');
            
            const isHidden = button.textContent.includes('anzeigen');

            if (isHidden) {
                solutionSpans.forEach(span => span.style.display = 'inline-block');
                button.textContent = 'Lösungen verbergen';
            } else {
                solutionSpans.forEach(span => span.style.display = 'none');
                button.textContent = 'Lösungen anzeigen';
            }
        }

        window.onload = () => {
             renderTFGrid();
        }
    </script>
